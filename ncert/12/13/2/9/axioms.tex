	Since
\begin{align}
A^{\prime}B^{\prime} &=  \brak{A+B}^{\prime},
\\
\pr{A^{\prime}B^{\prime}} &=  \pr{\brak{A+B}^{\prime}} 
\\
&= 1 - \pr{A+B} 
\label{eq:ncert/12/13/2/9/1}
\end{align} 
\iffalse
\item 
\begin{align}
 A+B &= A\brak{B+B^{\prime}} + B
\\
&= B \brak{A +1} + A B^{\prime}
\\
&=B + A B^{\prime}
\\
\implies \pr{A+B} &= \pr{B + A B^{\prime} }
\\
&=\pr{B}+\pr{ A B^{\prime} } 
\\
&B \brak{ A B^{\prime} } = 0
\label{eq:ncert/12/13/2/9/2}
\end{align}
\item 
\begin{align}
A = A \brak{B+B^{\prime}} =  AB + AB^{\prime}
\label{eq:ncert/12/13/2/9/3}
\end{align}
and 
\begin{align}
\brak{ AB}\brak{  AB^{\prime}} = 0, 
\\
 BB^{\prime} = 0
\label{eq:ncert/12/13/2/9/4}
\end{align}
Hence, $AB$ and $AB^{\prime}$ are mutually exclusive and 
%
\begin{align}
\pr{A} = \pr{AB} + \pr{AB^{\prime}}
\\
\implies 
\pr{AB^{\prime}} =  \pr{A} - \pr{AB}
\label{eq:ncert/12/13/2/9/5}
\end{align}
\item Substituting \eqref{eq:ncert/12/13/2/9/5} in \eqref{eq:ncert/12/13/2/9/2}, 
\begin{align}
\pr{A+B} &= \pr{A} + \pr{B} - \pr{AB} 
\label{eq:ncert/12/13/2/9/6}
\end{align}
\item Substituting \eqref{eq:ncert/12/13/2/9/6} in \eqref{eq:ncert/12/13/2/9/1}
	\fi
	Thus,
\begin{align}
\pr{A^{\prime}B^{\prime}} &=  1 - \cbrak{\pr{A} + \pr{B} - \pr{AB} }
\\
&= \frac{3}{8}
\label{eq:ncert/12/13/2/9/7}
\end{align}
